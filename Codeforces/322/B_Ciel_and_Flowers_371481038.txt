a,b,c = map(int , input().split())
 
ans1 =(a//3)+(b//3)+(c//3) +min(a%3,b%3,c%3)
ans2 = [0]
i = 1
anss = min(a,b,c)
while i<=2:
  if(min(a,b,c)>=i):
    if ans2[0]<(((a-i)//3)+((b-i)//3)+((c-i)//3)+i):
      ans2[0] = ((a-i)//3)+((b-i)//3)+((c-i)//3)+i
  else:
    break
  i+=1
ansss = min(a,b,c)
a,b,c = a-min(a,b,c), b-min(a,b,c), c-min(a,b,c)
ansss = ansss+(a//3)+(b//3)+(c//3)
print(max(ans1, max(ans2), ansss, anss))
