import sys
input = sys.stdin.readline

def linp(a):
    return list(map(a, input().split()))
def sinp():
    return input().strip()
def iinp():
    return int(input())

def m(l):
    return max(l)
def absinp():
    return [abs(x) for x in linp(int)]
def prefixsum(l):
    ps = l[:]
    for i in range(1, len(ps)):
        ps[i] += ps[i-1]
    return ps
