from collections import Counter

for i in range(int(input())):
  a = int(input())
  b = list(map(int, input().split()))
  c = Counter(b).most_common()
  rem = 0
  for j in range(len(c)):
    if c[j][0] != c[j][1] and c[j][0]<=c[j][1]:
      rem+= abs(c[j][0]-c[j][1])
    elif c[j][0]>c[j][1]:
      rem+=c[j][1]
  print(rem)
